\section*{Collect}

\begin{latinenglishsection}

\latinenglish{
	\Vbar.~Dóminus vobíscum.\\
	\Rbar.~Et cum spíritu tuo.
	
	Orémus.
	Omnípotens et miséricors Deus, univérsa nobis adversántia propitiátus exclúde:
	ut mente et córpore páriter expedíti, quæ tua sunt, líberis méntibus exsequámur.

	The Lord be with you.
	\Rbar.~And with thy spirit.
	
	Let us pray.
	O almighty and merciful God, in Thy goodness keep us, we beseech Thee, from all things hurtful;

	Through our Lord.
	\Rbar.~Amen.
}

\end{latinenglishsection}

\section*{Commemoration of the Holy Guardian Angels}

\textit{\textnormal{Ant.} Holy Angels, our Guardians, defend us in battle, that we may not perish in the terrible judgment.}

\gresetinitiallines{1}
\gregorioscore{../commemorations/guardian-angels-2v}

\begin{latinenglishsection}

\latinenglish{
	\Vbar.~In conspéctu Angelórum psallam tibi Deus meus.\\
	\Rbar.~Adorábo ad templum sanctum tuum, et confitébor nómini tuo.
	
	Orémus.
	Deus, qui ineffábili providéntia sanctos Angelos tuos ad nostram custódiam míttere dignáris:
largíre supplícibus tuis; et eórum semper protectióne deféndi, et ætérna societáte gaudére.
}{
	In the sight of the Angels I will sing unto Thee, O my God.
	\Rbar.~I will worship at Thy holy temple, and give praise to Thy Name.
	
	Let us pray.
	O God, Who in Thine ineffable providence hast deigned to send Thy holy Angels to watch over us,
	grant to Thy suppliants always to find safety in their protection and in eternity to share their happiness.
}

\end{latinenglishsection}

\section*{Commemoration of St. Therese of the Child Jesus}

\textit{\textnormal{Ant.} Come, Spouse of Christ, receive the crown which the Lord hath prepared for thee for all eternity.}

\gregorioscore{../commemorations/virgin-1v}

\begin{latinenglishsection}

\latinenglish{
	\Vbar.~Spécia tua et pulchritúdine tua.\\
	\Rbar.~Inténde, próspere procéde, et regna.
	
	Orémus.
	Dómine, qui dixisti: Nisi efficiámini sicut párvuli, non intrábitis in regnum c{\oe}lórum:
	da nobis, qu\'{\ae}sumus: ita sanctæ Terésiæ Vírginis in humilitáte et simplicitáte cordis vestígia sectári,
	ut præmia consequámur ætérna.
	Qui vivis et regnas.
	\Rbar.~Amen.
}{
	In thy glory and thy splendor.
	\Rbar.~Go forth, advance with victory and reign.
	
	Let us pray.
	O Lord, Who hast said: Unless ye become as little children ye shall not enter into the kingdom of heaven:
	grant us, we beseech Thee, so to follow in humility and simplicity of heart, the footsteps of the virgin St. Teresa,
	that we may obtain everlasting reward.
	Who livest and reignest.
	\Rbar.~Amen.
}

\end{latinenglishsection}
